1，列表的定义，共有3种方式
<html>
    <head>
         <title>study</title>
    </head>
    <body>
        <!-- ol是有序列表，ul是无序列表，li元素表示列表项 -->
       <ol>
           <li>charge Segway</li>
           <li>my favorite digital products
               <ul>
                   <li>mobile phone</li>
                   <li>pad</li>
                   <li>ipod</li>    
               </ul>
           </li>
       </ol>
       <!-- dl是定义列表，dt是列表项，dd是对列表项的描述 -->
       <dl>
           <dt>source of water</dt>
           <dd>
               we all know that:China has the thirst water 
           </dd>
       </dl>
    </body>
</html>
